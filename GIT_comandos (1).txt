---clonar repositorio
git clone https://..............

---iniciar git
git init 

-------unirte a una rama
git checkout nombre_rama

---uniser a rama remota 
git checkout -b nombra_rama 

----traer datos de una rama
git pull origin nombre_rama

----ver cambios realizados
git status

----agregar todos los cambios
git add .

-----realizar commit
git commit -m "nombre_rama comentarios"

---publicar commit
git push origin #rama
